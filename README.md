a = int(input("Сколько билетов вы хотите приобрести? "))
print("Укажите поочередно возраст посетителей ")
b = 0
for i in range(a):
    age = int(input("Возраст : "))
    if age < 18:
        b += 0
    elif 18 <= age < 25:
        b += 990
    else:
        b += 1390

if a > 3:
        print("Полная соимость билетов :", b ,"Pуб.")
        b *= 0.9
        print("Стоимость билетов сo скидкой:", b,"Pуб.")
else:
        print("Полная соимость билетов :", b ,"Pуб.")
